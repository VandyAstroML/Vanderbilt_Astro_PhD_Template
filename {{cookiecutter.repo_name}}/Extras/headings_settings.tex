%%
%% Page Settings
\textwidth=6in \oddsidemargin=0.5in \topmargin=-0.5in
\textheight=9in  % 9in must include page numbers

%%
%% Graphics and Figures

\DeclareGraphicsExtensions{.pdf, .jpeg, .png, .PNG, .eps, .tiff}

\urlstyle{same}
